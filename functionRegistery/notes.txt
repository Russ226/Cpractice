	int main(void) {
		void *handle = dlopen("./libconfig.so", RTLD_NOW);
		if (!handle) { fprintf(stderr, "dlopen failed: %s\n", dlerror()); return 1; }

		dlerror();
		int *global_limit = (int *)dlsym(handle, "global_limit");
		const char *err = dlerror();
		if (err) { fprintf(stderr, "dlsym failed: %s\n", err); dlclose(handle); return 1; }

		printf("global_limit = %d\n", *global_limit);
		dlclose(handle);
		return 0;
	}
